import java.io.BufferedReader;  
import java.io.InputStreamReader;  
import java.util.HashMap;  
import java.util.Map;  
import java.util.StringTokenizer;  
  
public class Main {  
  
    static int T, N, M;  
    static int[] A, B;  
  
    public static void main(String[] args) throws Exception {  
        init();  
        Map<Integer, Integer> map = new HashMap<>();  
        for (int n = 0; n < N; n++) {  
            int sum = 0;  
            for (int i = n; i < N; i++) {  
                sum += A[i];  
                if (map.containsKey(sum)) {  
                    map.replace(sum, map.get(sum) + 1);  
                } else {  
                    map.put(sum, 1);  
                }  
            }  
        }  
  
        long answer = 0;  
        for (int m = 0; m < M; m++) {  
            int sum = 0;  
            for (int i = m; i < M; i++) {  
                sum += B[i];  
                if (map.containsKey(T - sum)) {  
                    answer += map.get(T - sum);  
                }  
            }  
        }  
        System.out.println(answer);  
    }  
  
    static void init() throws Exception {  
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
        T = Integer.parseInt(br.readLine().trim());  
  
        N = Integer.parseInt(br.readLine().trim());  
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");  
        A = new int[N];  
        for (int n = 0; n < N; n++) {  
            A[n] = Integer.parseInt(st.nextToken());

        st = new StringTokenizer(br.readLine(), " ");

import java.io.BufferedReader;  
import java.io.InputStreamReader;  
import java.util.*;  
  
public class Main {  
  
    static int T, N, M;  
    static List<long[]> sumA;  
    static List<long[]> sumB;  
  
    public static void main(String[] args) throws Exception {  
        init();  
        long answer = 0;  
        for (long[] nodeA : sumA) {  
            answer += (findPair(T - nodeA[0]) * nodeA[1]);  
        }  
        System.out.println(answer);  
    }  
  
    static void init() throws Exception {  
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
        T = Integer.parseInt(br.readLine().trim());  
  
        N = Integer.parseInt(br.readLine().trim());  
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");  
        PriorityQueue<Long> pq = new PriorityQueue<>();  
        long[] A = new long[N + 1];  
        for (int n = 1; n <= N; n++) {  
            A[n] = A[n - 1] + Integer.parseInt(st.nextToken());  
            for (int a = n - 1; a >= 0; a--) {  
                pq.offer(A[n] - A[a]);  
            }  
        }  
        sumA = new ArrayList<>();  
        while(!pq.isEmpty()){  
            long target = pq.poll();  
            long count = 1;  
            while(!pq.isEmpty()){  
                long tmp = pq.poll();  
                if(tmp != target){  
                    pq.offer(tmp);  
                    break;                }  
                count++;  
            }  
            sumA.add(new long[]{target, count});  
        }  
  
        M = Integer.parseInt(br.readLine().trim());  
        st = new StringTokenizer(br.readLine(), " ");  
        long[] B = new long[M + 1];  
        sumB = new LinkedList<>();  
        for (int m = 1; m <= M; m++) {  
            B[m] = B[m - 1] + Integer.parseInt(st.nextToken());  
            for (int b = m - 1; b >= 0; b--) {  
                pq.offer(B[m] - B[b]);  
            }  
        }  
        sumB = new ArrayList<>();  
        while(!pq.isEmpty()){  
            long target = pq.poll();  
            long count = 1;  
            while(!pq.isEmpty()){  
                long tmp = pq.poll();  
                if(tmp != target){  
                    pq.offer(tmp);  
                    break;                }  
                count++;  
            }  
            sumB.add(new long[]{target, count});  
        }  
    }  
  
    static long findPair(long target) {  
        int start = 0, end = sumB.size() - 1;  
        while (start <= end) {  
            int mid = (start + end) / 2;  
            if (sumB.get(mid)[0] == target) {  
                return sumB.get(mid)[1];  
            } else if (sumB.get(mid)[0] < target) {  
                start = mid + 1;  
            } else {  
                end = mid - 1;  
            }  
        }  
        return 0;  
    }  
  
}
