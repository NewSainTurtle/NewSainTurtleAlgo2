import java.io.*;
import java.util.*;

class Solution {
    public String solution(String play_time, String adv_time, String[] logs) {
        int playTime = timeToSec(play_time); // ì¬ìƒì‹œê°„
        int advTime = timeToSec(adv_time); // ê´‘ê³ ì‹œê°„
        
        // ëˆ„ì í•©
        int maxLogTime = 0;
        int[] view = new int[playTime+1];
        for(String log:logs){
            String[] str = log.split("-");
            int startTime = timeToSec(str[0]);
            int endTime = timeToSec(str[1]);
            view[startTime]++;
            view[endTime]--;
            maxLogTime = Integer.max(maxLogTime,endTime);
        }
        
        for(int i=1;i<=maxLogTime;i++){
            view[i] += view[i-1];
        }
        
        // êµ¬ê°„í•©
        long sum = 0;
        for(int i=0;i<advTime;i++){
            sum+=view[i];
        }
        
        int resultTime = 0;
        long maxView=sum;
        for(int start=0, end=advTime; end<=maxLogTime; start++, end++){
            sum+=view[end] - view[start];
            if(sum>maxView) {
                maxView = sum;
                resultTime = start+1;
            }
        }
        
        return secToTime(resultTime);
    }
    
    static int timeToSec(String timeStr){
        int time=0;
        String[] str = timeStr.split(":");
        time += Integer.parseInt(str[0])*60*60; 
        time += Integer.parseInt(str[1])*60; 
        time += Integer.parseInt(str[2]);
        return time;
    }
    
    static String secToTime(int time){
        StringBuilder sb= new StringBuilder();
        int h = time/60/60;
        int m = time/60%60;
        int s = time%60%60;
        
        if(h<10) sb.append('0');
        sb.append(h).append(':');
        if(m<10) sb.append('0');
        sb.append(m).append(':');
        if(s<10) sb.append('0');
        sb.append(s);
                 
        return sb.toString();
    }
}
