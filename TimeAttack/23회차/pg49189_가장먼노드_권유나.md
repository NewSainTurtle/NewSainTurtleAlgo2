import java.util.*;

class Solution {
    final static int MAX = 50001;
    int N;
    int[] checked;
    boolean[] visited;
    
    public int solution(int n, int[][] edge) {
        N = n;
        checked = new int[N];
        visited = new boolean[N];
        Arrays.fill(checked, MAX);
        checked[0] = 0;
        
        for(int i=0; i < N-1; i++){
            int v = findVertex();
            visited[v] = true;
        
            for(int[] e : edge){
                if(e[0]-1 == v && !visited[e[1]-1] && (checked[e[0]-1]+1) < checked[e[1]-1]){
                    checked[e[1]-1] = checked[e[0]-1]+1;
                }else if(e[1]-1 == v && !visited[e[0]-1] && (checked[e[1]-1]+1) < checked[e[0]-1]){
                    checked[e[0]-1] = checked[e[1]-1]+1;
                }
            }
        }
        
        int max = 0;
        int count = 0;
        for(int i=0; i < N; i++){
            if(max < checked[i]){
                max = checked[i];
                count = 1;
            }else if(checked[i] == max){
                count++;
            }
        }
        
        return count;
    }
    
    int findVertex(){
        int min = MAX;
        int target = -1;
        
        for(int i=0; i < N; i++){
            if(!visited[i] && checked[i] < min){
                min = checked[i];
                target = i;
            }
        }
        return target;
    }
}

import java.util.List;
import java.util.ArrayList;
import java.util.Arrays;

class Solution {
    final static int MAX = 50001;
    int N;
    int[] checked;
    List<Integer>[] edges;
    boolean[] visited;
    
    public int solution(int n, int[][] edge) {
        init(n, edge);
        checkedMinDist();
        return countMaxDistNode();
    }
    
    void init(int n, int[][] edge){
        edges = new ArrayList[n];
        for(int i=0; i<n; i++){
            edges[i] = new ArrayList<>();
        }
        
        for(int[] e : edge){
            edges[e[0]-1].add(e[1]-1);
            edges[e[1]-1].add(e[0]-1);
        }
        
        N = n;
    }
    
    void checkedMinDist(){
        checked = new int[N];
        visited = new boolean[N];
        Arrays.fill(checked, MAX);
        checked[0] = 0;
        
        for(int i=0; i < N-1; i++){
            int v = findVertex();
            visited[v] = true;
        
            for(int cost : edges[v]){
                if(!visited[cost] && checked[v]+1 < checked[cost]){
                    checked[cost] = checked[v]+1;
                }
            }
        }
        
    }
    
    int findVertex(){
        int min = MAX;
        int target = -1;
        
        for(int i=0; i < N; i++){
            if(!visited[i] && checked[i] < min){
                min = checked[i];
                target = i;
            }
        }
        return target;
    }
    
    int countMaxDistNode(){
        int max = 0;
        int count = 0;
        for(int i=0; i < N; i++){
            if(max < checked[i]){
                max = checked[i];
                count = 1;
            }else if(checked[i] == max){
                count++;
            }
        }
        
        return count;
    }
}
