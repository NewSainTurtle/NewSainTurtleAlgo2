import java.io.*;
import java.util.*;

class Solution {
    public String solution(int n, int k, String[] cmd) {
        int now = k; // í˜„ì¬ ì„ íƒëœ í–‰
        int size = n-1;
        Stack<Integer> stack = new Stack();
        
        for(String str: cmd){
            StringTokenizer st = new StringTokenizer(str);
            char c = st.nextToken().charAt(0); // ëª…ë ¹ì–´
            if(c=='U'){
                now -= Integer.parseInt(st.nextToken());
            } else if(c=='D') { 
                now += Integer.parseInt(st.nextToken());
            } else if(c=='C'){
                stack.add(now);
                if(now==size) now--;
                size--;
            } else if(c=='Z') {
                if(stack.pop()<=now) now++;
                size++;
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for(int i=0;i<=size;i++){
            sb.append('O');
        }

        while(!stack.isEmpty()){
            int temp = stack.pop();
            sb.insert(temp,'X');
        }
        
        return sb.toString();
    }
}
