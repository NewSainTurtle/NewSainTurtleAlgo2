import java.io.*;
import java.util.*;

class Solution {
    static int[] disconts = {10,20,30,40};  // ì´ëª¨í‹°ì½˜ í• ì¸ë¥ 
    static int maxCnt = 0, maxPrice = 0; // ìµœëŒ€ í”ŒëŸ¬ìŠ¤ ì´ìš©ì, ìµœëŒ€ íŒë§¤ì•¡
    static int N,M;
    
    public int[] solution(int[][] users, int[] emoticons) {
        N = users.length;
        M = emoticons.length;
        getPer(users, emoticons, 0, new int[M]); // ê°€ëŠ¥í•œ ì´ëª¨í‹°ì½˜ í• ì¸ë¥  ì¡°í•© êµ¬í•˜ê¸°
        return new int[]{maxCnt, maxPrice};

    private static void caclPrice(int[][] users, int[] emoticons, int[] disArr) {
        int cnt = 0;    // ì´ëª¨í‹°ì½˜ í”ŒëŸ¬ìŠ¤ êµ¬ë§¤ì ìˆ˜
        int sum = 0;    // ì´ëª¨í‹°ì½˜ ì´ íŒë§¤ì•¡
        
        for (int[] user : users) {
            int price = 0;  // í•œ ì‚¬ëŒì´ êµ¬ë§¤í•œ ì´ëª¨í‹°ì½˜ ê°€ê²©
            for(int i=0; i<disArr.length; i++){
                if(disArr[i] >= user[0]) {  // í• ì¸ë¥  ì´ìƒì´ë©´
                    price += emoticons[i] * (100 - disArr[i]) / 100;
                }
                if (price >= user[1]) {  // ê¸°ì¤€ ê°€ê²© ì´ìƒì´ ëœë‹¤ë©´
                    cnt++;
                    price = 0;
                    break; // í”ŒëŸ¬ìŠ¤ êµ¬ë§¤ìì´ë¯€ë¡œ ë‹¤ìŒ ìœ ì € íƒìƒ‰
                }
            }
            sum += price;
        }
            
        if(cnt > maxCnt) {
             maxCnt = cnt;
             maxPrice = sum;
        } else if(cnt == maxCnt){
             maxPrice = Math.max(maxPrice, sum);
        }
    }
    
     // ì¤‘ë³µ ì¡°í•©
    private static void getPer(int[][] users, int[] emoticons, int depth, int[] disArr) {
		if (depth == M) {
            caclPrice(users, emoticons, disArr);
			return;
		}
		
		for (int rate : disconts) {
			disArr[depth] = rate;
            getPer(users, emoticons, depth + 1, disArr);
		}
	}           
    
}
